    \begin{aligned}
      \mat{A} &=
      \begin{bmatrix}
        a_{11} & \cdots & a_{1n} \\
        \cdot  & \cdots & \cdot  \\

      \end{bmatrix}
    \end{aligned}
    \begin{aligned}
      \vec{g} &=
      \begin{bmatrix}
        g_{1}  \\
        \vdots \\
        g_{n}
      \end{bmatrix}
    \end{aligned}
  \end{split}
